1//! HAL interface to the AES electronic codebook mode encryption.
2//!
3//! The ECB encryption block supports 128 bit AES encryption (encryption only, not decryption).
4
5#[cfg(not(feature = "5340-net"))]
6use crate::pac::ECB;
7#[cfg(feature = "5340-net")]
8use crate::pac::ECB_NS as ECB;
9
10use core::sync::atomic::{compiler_fence, Ordering};
11
12/// Error type to represent a sharing conflict during encryption.
13#[derive(Debug, Copy, Clone)]
14pub struct EncryptionError {}
15
16/// A safe, blocking wrapper around the AES-ECB peripheral.
17///
18/// It's really just blockwise AES and not an ECB stream cipher. Blocks can be
19/// encrypted by calling `crypt_block`.
20pub struct Ecb {
21    regs: ECB,
22}
23
24impl Ecb {
25    /// Takes ownership of the `ECB` peripheral, returning a safe wrapper.
26    pub fn init(regs: ECB) -> Self {
27        // Disable all interrupts
28        regs.intenclr
29            .write(|w| w.endecb().clear().errorecb().clear());
30
31        // NOTE(unsafe) 1 is a valid pattern to write to this register.
32        regs.tasks_stopecb.write(|w| unsafe { w.bits(1) });
33        Self { regs }
34    }
35
36    /// Destroys `self`, giving the `ECB` peripheral back.
37    pub fn into_inner(self) -> ECB {
38        // Clear all events
39        self.regs.events_endecb.reset();
40        self.regs.events_errorecb.reset();
41
42        self.regs
43    }
44
45    /// Blocking encryption.
46    ///
47    /// Encrypts a `block` with `key`.
48    ///
49    /// # Errors
50    ///
51    /// An error will be returned when the AES hardware raises an `ERRORECB`
52    /// event. This can happen when an operation is started that shares the AES
53    /// hardware resources with the AES ECB peripheral while an encryption
54    /// operation is running.
55    pub fn encrypt_block(
56        &mut self,
57        block: [u8; 16],
58        key: [u8; 16],
59    ) -> Result<[u8; 16], EncryptionError> {
60        #[repr(C)]
61        struct EcbData {
62            key: [u8; 16],
63            clear_text: [u8; 16],
64            cipher_text: [u8; 16],
65        }
66
67        // We allocate the DMA'd buffer on the stack, which means that we must
68        // not panic or return before the AES operation is finished.
69        let mut buf = EcbData {
70            key,
71            clear_text: block,
72            cipher_text: [0; 16],
73        };
74
75        // NOTE(unsafe) Any 32bits pattern is safe to write to this register.
76        self.regs
77            .ecbdataptr
78            .write(|w| unsafe { w.bits(&mut buf as *mut _ as u32) });
79
80        // Clear all events.
81        self.regs.events_endecb.reset();
82        self.regs.events_errorecb.reset();
83
84        // "Preceding reads and writes cannot be moved past subsequent writes."
85        compiler_fence(Ordering::Release);
86        // NOTE(unsafe) 1 is a valid pattern to write to this register
87        self.regs.tasks_startecb.write(|w| unsafe { w.bits(1) });
88
89        while self.regs.events_endecb.read().bits() == 0
90            && self.regs.events_errorecb.read().bits() == 0
91        {}
92
93        // "Subsequent reads and writes cannot be moved ahead of preceding reads."
94        compiler_fence(Ordering::Acquire);
95
96        if self.regs.events_errorecb.read().bits() == 1 {
97            // It's ok to return here, the events will be cleared before the next encryption.
98            return Err(EncryptionError {});
99        }
100        Ok(buf.cipher_text)
101    }
102}
