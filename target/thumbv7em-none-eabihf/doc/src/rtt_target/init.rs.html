13/// rtt_init! implementation detail
14#[macro_export]
15#[doc(hidden)]
16macro_rules! rtt_init_channels {
17    (
18        $field:expr;
19        $number:literal: {
20            $(
21                size: $size:expr
22                $(, mode: $mode:path )?
23                $(, name: $name:literal )?
24                $(, section: $section:literal )?
25                $(,)?
26            )?
27        }
28        $($tail:tt)*
29    ) => {
30        $(
31            let mut name: *const u8 = core::ptr::null();
32            $( name = concat!($name, "\0").as_bytes().as_ptr(); )?
33
34            let mut mode = $crate::ChannelMode::NoBlockSkip;
35            $( mode = $mode; )?
36
37            $field[$number].init(name, mode, {
38                $( #[link_section = $section] )?
39                static mut _RTT_CHANNEL_BUFFER: MaybeUninit<[u8; $size]> = MaybeUninit::uninit();
40                _RTT_CHANNEL_BUFFER.as_mut_ptr()
41            });
42        )?
43
44        $crate::rtt_init_channels!($field; $($tail)*);
45    };
46    ($field:expr;) => { };
47}

68/// Initializes RTT with the specified channels. Channel numbers, buffer sizes and names can be
69/// defined.
70///
71/// It is also possible to pre-allocate uninitialized channels to let crates accessing RTT via FFI
72/// play nicely with crates using rtt-target directly.
73///
74/// The syntax looks as follows (note that commas are not allowed anywhere):
75///
76/// ```
77/// let channels = rtt_init! {
78///     up: {
79///         0: { // channel number
80///             size: 1024, // buffer size in bytes
81///             mode: NoBlockSkip, // mode (optional, default: NoBlockSkip, see enum ChannelMode)
82///             name: "Terminal", // name (optional, default: no name)
83///             section: ".segger_term_buf" // Buffer linker section (optional, default: no section)
84///         }
85///         1: {
86///             size: 32
87///         }
88///         2: { } // pre-allocated channel without buffer
89///     }
90///     down: {
91///         0: {
92///             size: 16,
93///             name: "Terminal"
94///         }
95///     }
96///     section_cb: ".segger_rtt" // Control block linker section (optional, default: no section)
97/// };
98/// ```
99///
100/// The channel numbers must start from 0 and not skip any numbers, or otherwise odd things will
101/// happen. The order of the channel parameters is fixed, but optional parameters can be left out.
102///
103/// If no size is given, then an RTT channel header entry will be allocated with its buffer pointer
104/// set to `null`. This allows third-party FFI crates (e.g. [`rtos-trace`][rtos_trace]/
105/// [`systemview-target`][systemview_target]) to identify and initialize the channel with its own
106/// name and buffer from C code.
107///
108/// Note: Rust crates should rely on channels initialized through `rtt_init!`. That's why this crate
109/// does not provide an API to initialize pre-allocated channels from Rust.
110///
111/// This macro should be called once within a function, preferably close to the start of your entry
112/// point. The macro must only be called once - if it's called twice in the same program a duplicate
113/// symbol error will occur.
114///
115/// At compile time the macro will statically reserve space for the RTT control block and the
116/// channel buffers. At runtime the macro fills in the structures and prepares them for use.
117///
118/// The macro returns a generate struct that contains the channels. The struct for the example above
119/// would look as follows:
120///
121/// ```
122/// struct Channels {
123///     up: (UpChannel, UpChannel),
124///     down: (DownChannel,),
125/// }
126/// ```
127///
128/// The channels can either be accessed by reference or moved out as needed. For example:
129///
130/// ```
131/// use core::fmt::Write;
132///
133/// let channels = rtt_init! { ... };
134/// let mut output = channels.up.0;
135/// writeln!(output, "Hello, world!").ok();
136/// ```
137///
138/// [rtos_trace]: https://docs.rs/rtos-trace
139/// [systemview_target]: https://docs.rs/systemview-target/
140#[macro_export]
141macro_rules! rtt_init {
142    {
143        $(up: { $($up:tt)* } )?
144        $(down: { $($down:tt)* } )?
145        $(section_cb: $section_cb:literal )?
146    } => {{
147        use core::mem::MaybeUninit;
148        use core::ptr;
149        use core::cell::Cell;
150        use $crate::UpChannel;
151        use $crate::DownChannel;
152        use $crate::rtt::*;
153
154        #[repr(C)]
155        pub struct RttControlBlock {
156            header: RttHeader,
157            up_channels: [RttChannel; $crate::rtt_init_repeat!({ 1 + } { 0 }; $($($up)*)?)],
158            down_channels: [RttChannel; $crate::rtt_init_repeat!({ 1 + } { 0 }; $($($down)*)?)],
159        }
160
161        #[used]
162        #[no_mangle]
163        #[export_name = "_SEGGER_RTT"]
164        $( #[link_section = $section_cb] )?
165        pub static mut CONTROL_BLOCK: MaybeUninit<RttControlBlock> = MaybeUninit::uninit();
166
167        #[allow(unused)]
168        #[export_name = "rtt_init_must_not_be_called_multiple_times"]
169        fn rtt_init_must_not_be_called_multiple_times() { }
170
171        use ::rtt_target::export::critical_section;
172
173        static INITIALIZED: critical_section::Mutex<Cell<bool>> = critical_section::Mutex::new(Cell::new(false));
174        critical_section::with(|cs| {
175            if INITIALIZED.borrow(cs).get() {
176                panic!("rtt_init! must not be called multiple times");
177            }
178            INITIALIZED.borrow(cs).set(true);
179        });
180
181        unsafe {
182            ptr::write_bytes(CONTROL_BLOCK.as_mut_ptr(), 0, 1);
183
184            let cb = &mut *CONTROL_BLOCK.as_mut_ptr();
185
186            $( $crate::rtt_init_channels!(cb.up_channels; $($up)*); )?
187            $( $crate::rtt_init_channels!(cb.down_channels; $($down)*); )?
188
189            // The header is initialized last to make it less likely an unfinished control block is
190            // detected by the host.
191
192            cb.header.init(cb.up_channels.len(), cb.down_channels.len());
193
194            pub struct Channels {
195                $( pub up: $crate::rtt_init_repeat!({ UpChannel, } {}; $($up)*), )?
196                $( pub down: $crate::rtt_init_repeat!({ DownChannel, } {}; $($down)*), )?
197            }
198
199            Channels {
200                $( up: $crate::rtt_init_wrappers!(cb.up_channels; UpChannel::new; {}; $($up)*), )?
201                $( down: $crate::rtt_init_wrappers!(cb.down_channels; DownChannel::new; {}; $($down)*), )?
202            }
203        }
204    }};
205}

207/// Initializes RTT with default up/down channels.
208///
209/// The default channels are up channel 0 with a 1024 byte buffer and down channel 0 with a 16 byte
210/// buffer. Both channels are called "Terminal". This macro is equivalent to:
211///
212/// ```
213/// rtt_init! {
214///     up: {
215///         0: {
216///             size: 1024,
217///             name: "Terminal"
218///         }
219///     }
220///     down: {
221///         0: {
222///             size: 16,
223///             name: "Terminal"
224///         }
225///     }
226/// };
227/// ```
228///
229/// See [`rtt_init`] for more details.
230#[macro_export]
231macro_rules! rtt_init_default {
232    () => {
233        $crate::rtt_init! {
234            up: {
235                0: {
236                    size: 1024,
237                    name: "Terminal"
238                }
239            }
240            down: {
241                0: {
242                    size: 16,
243                    name: "Terminal"
244                }
245            }
246        };
247    };
