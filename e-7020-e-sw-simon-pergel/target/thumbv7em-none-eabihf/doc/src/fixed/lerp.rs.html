18macro_rules! make_lerp {
19    ($i:ident, $u:ident, $ii:ident, $uu:ident, $uu1:expr) => {
20        pub const fn $i(r: $i, start: $i, end: $i, frac_bits: u32) -> ($i, bool) {
21            // 0x00 ≤ r_abs ≤ 0x80
22            let (r_abs, r_neg) = if r >= 0 {
23                (r as $u, false)
24            } else {
25                (r.wrapping_neg() as $u, true)
26            };
27            // 0x00 ≤ range_abs ≤ 0xff
28            let (range_abs, range_neg) = if end >= start {
29                (end.wrapping_sub(start) as $u, false)
30            } else {
31                (start.wrapping_sub(end) as $u, true)
32            };
33            // 0x0000 ≤ wide_abs ≤ 0x7f80
34            let wide_abs = (r_abs as $uu) * (range_abs as $uu);
35            let wide_neg = r_neg != range_neg;
36            let wide_uns = if wide_neg {
37                // 0x0000 ≤ add ≤ 0x00ff
38                let add = ($uu1 << frac_bits) - 1;
39                // If frac_bits is 0: add = 0x0000; 0x0000 ≤ shifted ≤ 0x7f80
40                // If frac_bits is max: add = 0x00ff; 0x0000 ≤ shifted ≤ 0x0080
41                let shifted = (wide_abs + add) >> frac_bits;
42                shifted.wrapping_neg()
43            } else {
44                wide_abs >> frac_bits
45            };
46            let wide = wide_uns as $ii;
47            let (wide_ret, overflow1) = wide.overflowing_add(start as $ii);
48            let ret = wide_ret as $i;
49            let overflow2 = wide_ret != ret as $ii;
50            (ret, overflow1 | overflow2)
51        }
52
53        pub const fn $u(r: $u, start: $u, end: $u, frac_bits: u32) -> ($u, bool) {
54            // 0x00 ≤ range_abs ≤ 0xff
55            let (range_abs, range_neg) = if end >= start {
56                (end - start, false)
57            } else {
58                (start - end, true)
59            };
60            // 0x0000 ≤ wide_abs ≤  0xfe01
61            let wide_abs = (r as $uu) * (range_abs as $uu);
62            let (wide_ret, overflow1) = if range_neg {
63                // 0x0000 ≤ add ≤ 0x00ff
64                let add = ($uu1 << frac_bits) - 1;
65                // If frac_bits is 0: add = 0x0000; 0x0000 ≤ shifted ≤ 0xfe01
66                // If frac_bits is max: add = 0x00ff; 0x0000 ≤ shifted ≤ 0x00ff
67                let shifted = (wide_abs + add) >> frac_bits;
68                (start as $uu).overflowing_sub(shifted)
69            } else {
70                let shifted = wide_abs >> frac_bits;
71                (start as $uu).overflowing_add(shifted)
72            };
73            let ret = wide_ret as $u;
74            let overflow2 = wide_ret != ret as $uu;
75            (ret, overflow1 | overflow2)
76        }
77    };
78}

85pub const fn i128(r: i128, start: i128, end: i128, frac_bits: u32) -> (i128, bool) {
86    // 0x00 ≤ r_abs ≤ 0x80
87    let (r_abs, r_neg) = if r >= 0 {
88        (r as u128, false)
89    } else {
90        (r.wrapping_neg() as u128, true)
91    };
92    // 0x00 ≤ range_abs ≤ 0xff
93    let (range_abs, range_neg) = if end >= start {
94        (end.wrapping_sub(start) as u128, false)
95    } else {
96        (start.wrapping_sub(end) as u128, true)
97    };
98    // 0x0000 ≤ wide_abs ≤ 0x7f80
99    let wide_abs = int256::wide_mul_u128(r_abs, range_abs);
100    let wide_neg = r_neg != range_neg;
101    let wide_uns = if wide_neg {
102        // 0x0000 ≤ add ≤ 0x00ff
103        let add = if frac_bits == 128 {
104            u128::MAX
105        } else {
106            (1u128 << frac_bits) - 1
107        };
108        // If frac_bits is 0: add = 0x0000; 0x0000 ≤ shifted ≤ 0x7f80
109        // If frac_bits is max: add = 0x00ff; 0x0000 ≤ shifted ≤ 0x0080
110        let sum = int256::wrapping_add_u256_u128(wide_abs, add);
111        let shifted = int256::shl_u256_max_128(sum, frac_bits);
112        int256::wrapping_neg_u256(shifted)
113    } else {
114        int256::shl_u256_max_128(wide_abs, frac_bits)
115    };
116    let wide = int256::u256_wrapping_as_i256(wide_uns);
117    let (wide_ret, overflow1) = int256::overflowing_add_i256_i128(wide, start);
118    let ret = wide_ret.lo as i128;
119    let overflow2 = if ret < 0 {
120        wide_ret.hi != -1
121    } else {
122        wide_ret.hi != 0
123    };
124    (ret, overflow1 | overflow2)
125}

127pub const fn u128(r: u128, start: u128, end: u128, frac_bits: u32) -> (u128, bool) {
128    // 0x00 ≤ range_abs ≤ 0xff
129    let (range_abs, range_neg) = if end >= start {
130        (end - start, false)
131    } else {
132        (start - end, true)
133    };
134    // 0x0000 ≤ wide_abs ≤  0xfe01
135    let wide_abs = int256::wide_mul_u128(r, range_abs);
136    if range_neg {
137        // 0x0000 ≤ add ≤ 0x00ff
138        let add = if frac_bits == 128 {
139            u128::MAX
140        } else {
141            (1u128 << frac_bits) - 1
142        };
143        // If frac_bits is 0: add = 0x0000; 0x0000 ≤ shifted ≤ 0xfe01
144        // If frac_bits is max: add = 0x00ff; 0x0000 ≤ shifted ≤ 0x00ff
145        let sum = int256::wrapping_add_u256_u128(wide_abs, add);
146        let shifted = int256::shl_u256_max_128(sum, frac_bits);
147        int256::overflowing_sub_u128_u256(start, shifted)
148    } else {
149        let shifted = int256::shl_u256_max_128(wide_abs, frac_bits);
150        int256::overflowing_add_u128_u256(start, shifted)
151    }
152}

154#[cfg(test)]
155mod tests {
156    use crate::lerp;
157
158    #[test]
159    fn lerp_i8() {
160        assert_eq!(lerp::i8(0, -128, 127, 0), (-128, false));
161        assert_eq!(lerp::i8(0, 127, -128, 8), (127, false));
162        assert_eq!(lerp::i8(1, -128, 127, 1), (-1, false));
163        assert_eq!(lerp::i8(-64, -100, -110, 6), (-90, false));
164
165        assert_eq!(lerp::i8(2 << 2, 0, 50, 2), (100, false));
166        assert_eq!(lerp::i8(2 << 2, 0, 100, 2), (-56, true));
167        assert_eq!(lerp::i8((-1) << 2, 50, 0, 2), (100, false));
168        assert_eq!(lerp::i8((-1) << 2, 100, 0, 2), (-56, true));
169    }
170
171    #[test]
172    fn lerp_u8() {
173        assert_eq!(lerp::u8(0, 0, 255, 0), (0, false));
174        assert_eq!(lerp::u8(0, 255, 0, 8), (255, false));
175        assert_eq!(lerp::u8(1, 0, 255, 0), (255, false));
176        assert_eq!(lerp::u8(128, 255, 0, 7), (0, false));
177
178        assert_eq!(lerp::u8(2 << 2, 0, 100, 2), (200, false));
179        assert_eq!(lerp::u8(2 << 2, 0, 200, 2), (200 - 56, true));
180    }
181
182    #[test]
183    fn lerp_i128() {
184        assert_eq!(lerp::i128(0, -128, 127, 0), (-128, false));
185        assert_eq!(lerp::i128(0, 127, -128, 128), (127, false));
186        assert_eq!(lerp::i128(1, -128, 127, 1), (-1, false));
187        assert_eq!(lerp::i128(1 << 98, 127, -128, 100), (63, false));
188        assert_eq!(lerp::i128((-64) << 120, -100, -110, 126), (-90, false));
189
190        assert_eq!(
191            lerp::i128(2 << 2, 0, i128::MAX / 2, 2),
192            (i128::MAX / 2 * 2, false)
193        );
194        assert_eq!(
195            lerp::i128(2 << 2, 0, i128::MAX, 2),
196            (i128::MAX.wrapping_mul(2), true)
197        );
198        assert_eq!(
199            lerp::i128((-1) << 2, i128::MAX / 2, 0, 2),
200            (i128::MAX / 2 * 2, false)
201        );
202        assert_eq!(
203            lerp::i128((-1) << 2, i128::MAX, 0, 2),
204            (i128::MAX.wrapping_mul(2), true)
205        );
206    }
207
208    #[test]
209    fn lerp_u128() {
210        assert_eq!(lerp::u128(0, 0, 255, 0), (0, false));
211        assert_eq!(lerp::u128(0, 255, 0, 128), (255, false));
212        assert_eq!(lerp::u128(1, 0, 255, 0), (255, false));
213        assert_eq!(lerp::u128(128 << 120, 255, 0, 127), (0, false));
214
215        assert_eq!(
216            lerp::u128(2 << 2, 0, u128::MAX / 2, 2),
217            (u128::MAX / 2 * 2, false)
218        );
219        assert_eq!(
220            lerp::u128(2 << 2, 0, u128::MAX, 2),
221            (u128::MAX.wrapping_mul(2), true)
222        );
223    }
