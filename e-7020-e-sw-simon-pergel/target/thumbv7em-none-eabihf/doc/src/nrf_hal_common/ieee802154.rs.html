40/// Clear Channel Assessment method
41pub enum Cca {
42    /// Carrier sense
43    CarrierSense,
44    /// Energy Detection / Energy Above Threshold
45    EnergyDetection {
46        /// Energy measurements above this value mean that the channel is assumed to be busy.
47        /// Note the the measurement range is 0..0xFF - where 0 means that the received power was
48        /// less than 10 dB above the selected receiver sensitivity. This value is not given in dBm,
49        /// but can be converted. See the nrf52840 Product Specification Section 6.20.12.4
50        /// for details.
51        ed_threshold: u8,
52    },
53}

149    /// Initializes the radio for IEEE 802.15.4 operation
150    pub fn init<L, LSTAT>(radio: RADIO, _clocks: &'c Clocks<ExternalOscillator, L, LSTAT>) -> Self {
151        let mut radio = Self {
152            needs_enable: false,
153            radio,
154            _clocks: PhantomData,
155        };
156
157        // shortcuts will be kept off by default and only be temporarily enabled within blocking
158        // functions
159        radio.radio.shorts.reset();
160
161        // go to a known state
162        radio.disable();
163
164        // clear any event of interest to us
165        radio.radio.events_disabled.reset();
166        radio.radio.events_end.reset();
167        radio.radio.events_phyend.reset();
168
169        radio.radio.mode.write(|w| w.mode().ieee802154_250kbit());
170
171        // NOTE(unsafe) radio is currently disabled
172        unsafe {
173            radio.radio.pcnf0.write(|w| {
174                w.s1incl()
175                    .clear_bit() // S1 not included in RAM
176                    .plen()
177                    ._32bit_zero() // 32-bit zero preamble
178                    .crcinc()
179                    .include() // the LENGTH field (the value) also accounts for the CRC (2 bytes)
180                    .cilen()
181                    .bits(0) // no code indicator
182                    .lflen()
183                    .bits(8) // length = 8 bits (but highest bit is reserved and must be `0`)
184                    .s0len()
185                    .clear_bit() // no S0
186                    .s1len()
187                    .bits(0) // no S1
188            });
189
190            radio.radio.pcnf1.write(|w| {
191                w.maxlen()
192                    .bits(Packet::MAX_PSDU_LEN) // payload length
193                    .statlen()
194                    .bits(0) // no static length
195                    .balen()
196                    .bits(0) // no base address
197                    .endian()
198                    .clear_bit() // little endian
199                    .whiteen()
200                    .clear_bit() // no data whitening
201            });
202
203            // CRC configuration required by the IEEE spec: x**16 + x**12 + x**5 + 1
204            radio
205                .radio
206                .crccnf
207                .write(|w| w.len().two().skipaddr().ieee802154());
208            radio.radio.crcpoly.write(|w| w.crcpoly().bits(0x11021));
209            radio.radio.crcinit.write(|w| w.crcinit().bits(0));
210        }
211
212        // set default settings
213        radio.set_channel(DEFAULT_CHANNEL);
214        radio.set_cca(DEFAULT_CCA);
215        radio.set_sfd(DEFAULT_SFD);
216        radio.set_txpower(DEFAULT_TXPOWER);
217
218        radio
219    }

260    /// Sample the received signal power (i.e. the presence of possibly interfering signals)
261    /// within the bandwidth of the currently used channel for `sample_cycles` iterations.
262    /// Note that one iteration has a sample time of 128μs, and that each iteration produces the
263    /// average RSSI value measured during this sample time.
264    ///
265    /// Returns the *maximum* measurement recorded during sampling as reported by the hardware (not in dBm!).
266    /// The result can be used to find a suitable ED threshold for Energy Detection-based CCA mechanisms.
267    ///
268    /// For details, see Section 6.20.12.3 Energy detection (ED) of the PS.
269    /// RSSI samples are averaged over a measurement time of 8 symbol periods (128 μs).
270    pub fn energy_detection_scan(&mut self, sample_cycles: u32) -> u8 {
271        unsafe {
272            // Increase the time spent listening
273            self.radio.edcnt.write(|w| w.edcnt().bits(sample_cycles));
274        }
275
276        // ensure that the shortcut between READY event and START task is disabled before putting
277        // the radio into recv mode
278        self.radio.shorts.reset();
279        self.put_in_rx_mode();
280
281        // clear related events
282        self.radio.events_edend.reset();
283
284        // start energy detection sampling
285        self.radio
286            .tasks_edstart
287            .write(|w| w.tasks_edstart().set_bit());
288
289        loop {
290            if self.radio.events_edend.read().events_edend().bit_is_set() {
291                // sampling period is over; collect value
292                self.radio.events_edend.reset();
293
294                // note that since we have increased EDCNT, the EDSAMPLE register contains the
295                // maximum recorded value, not the average
296                let read_lvl = self.radio.edsample.read().edlvl().bits();
297                return read_lvl;
298            }
299        }
300    }

302    /// Receives one radio packet and copies its contents into the given `packet` buffer
303    ///
304    /// This methods returns the `Ok` variant if the CRC included the packet was successfully
305    /// validated by the hardware; otherwise it returns the `Err` variant. In either case, `packet`
306    /// will be updated with the received packet's data
307    pub fn recv(&mut self, packet: &mut Packet) -> Result<u16, u16> {
308        // Start the read
309        // NOTE(unsafe) We block until reception completes or errors
310        unsafe {
311            self.start_recv(packet);
312        }
313
314        // wait until we have received something
315        self.wait_for_event(Event::End);
316        dma_end_fence();
317
318        let crc = self.radio.rxcrc.read().rxcrc().bits() as u16;
319        if self.radio.crcstatus.read().crcstatus().bit_is_set() {
320            Ok(crc)
321        } else {
322            Err(crc)
323        }
324    }

326    /// Listens for a packet for no longer than the specified amount of microseconds
327    /// and copies its contents into the given `packet` buffer
328    ///
329    /// If no packet is received within the specified time then the `Timeout` error is returned
330    ///
331    /// If a packet is received within the time span then the packet CRC is checked. If the CRC is
332    /// incorrect then the `Crc` error is returned; otherwise the `Ok` variant is returned.
333    /// Note that `packet` will contain the packet in any case, even if the CRC check failed.
334    ///
335    /// Note that the time it takes to switch the radio to RX mode is included in the timeout count.
336    /// This transition may take up to a hundred of microseconds; see the section 6.20.15.8 in the
337    /// Product Specification for more details about timing
338    pub fn recv_timeout<I>(
339        &mut self,
340        packet: &mut Packet,
341        timer: &mut Timer<I>,
342        microseconds: u32,
343    ) -> Result<u16, Error>
344    where
345        I: timer::Instance,
346    {
347        // Start the timeout timer
348        timer.start(microseconds);
349
350        // Start the read
351        // NOTE(unsafe) We block until reception completes or errors
352        unsafe {
353            self.start_recv(packet);
354        }
355
356        // Wait for transmission to end
357        let mut recv_completed = false;
358
359        loop {
360            if self.radio.events_end.read().bits() != 0 {
361                // transfer complete
362                dma_end_fence();
363                recv_completed = true;
364                break;
365            }
366
367            if timer.reset_if_finished() {
368                // timeout
369                break;
370            }
371        }
372
373        if !recv_completed {
374            // Cancel the reception if it did not complete until now
375            self.cancel_recv();
376            Err(Error::Timeout)
377        } else {
378            let crc = self.radio.rxcrc.read().rxcrc().bits() as u16;
379            if self.radio.crcstatus.read().crcstatus().bit_is_set() {
380                Ok(crc)
381            } else {
382                Err(Error::Crc(crc))
383            }
384        }
385    }

387    unsafe fn start_recv(&mut self, packet: &mut Packet) {
388        // NOTE we do NOT check the address of `packet` because the mutable reference ensures it's
389        // allocated in RAM
390
391        // clear related events
392        self.radio.events_phyend.reset();
393        self.radio.events_end.reset();
394
395        self.put_in_rx_mode();
396
397        // NOTE(unsafe) DMA transfer has not yet started
398        // set up RX buffer
399        self.radio
400            .packetptr
401            .write(|w| w.packetptr().bits(packet.buffer.as_mut_ptr() as u32));
402
403        // start transfer
404        dma_start_fence();
405        self.radio.tasks_start.write(|w| w.tasks_start().set_bit());
406    }

408    fn cancel_recv(&mut self) {
409        self.radio.tasks_stop.write(|w| w.tasks_stop().set_bit());
410        self.wait_for_state_a(STATE_A::RX_IDLE);
411        // DMA transfer may have been in progress so synchronize with its memory operations
412        dma_end_fence();

415    /// Tries to send the given `packet`
416    ///
417    /// This method performs Clear Channel Assessment (CCA) first and sends the `packet` only if the
418    /// channel is observed to be *clear* (no transmission is currently ongoing), otherwise no
419    /// packet is transmitted and the `Err` variant is returned
420    ///
421    /// NOTE this method will *not* modify the `packet` argument. The mutable reference is used to
422    /// ensure the `packet` buffer is allocated in RAM, which is required by the RADIO peripheral
423    // NOTE we do NOT check the address of `packet` because the mutable reference ensures it's
424    // allocated in RAM
425    pub fn try_send(&mut self, packet: &mut Packet) -> Result<(), ()> {
426        // enable radio to perform cca
427        self.put_in_rx_mode();
428
429        // clear related events
430        self.radio.events_phyend.reset();
431        self.radio.events_end.reset();
432
433        // NOTE(unsafe) DMA transfer has not yet started
434        unsafe {
435            self.radio
436                .packetptr
437                .write(|w| w.packetptr().bits(packet.buffer.as_ptr() as u32));
438        }
439
440        // configure radio to immediately start transmission if the channel is idle
441        self.radio.shorts.modify(|_, w| {
442            w.ccaidle_txen()
443                .set_bit()
444                .txready_start()
445                .set_bit()
446                .end_disable()
447                .set_bit()
448        });
449
450        // the DMA transfer will start at some point after the following write operation so
451        // we place the compiler fence here
452        dma_start_fence();
453        // start CCA. In case the channel is clear, the data at packetptr will be sent automatically
454        self.radio
455            .tasks_ccastart
456            .write(|w| w.tasks_ccastart().set_bit());
457
458        loop {
459            if self.radio.events_phyend.read().events_phyend().bit_is_set() {
460                // transmission completed
461                dma_end_fence();
462                self.radio.events_phyend.reset();
463                self.radio.shorts.reset();
464                return Ok(());
465            }
466
467            if self
468                .radio
469                .events_ccabusy
470                .read()
471                .events_ccabusy()
472                .bit_is_set()
473            {
474                // channel is busy
475                self.radio.events_ccabusy.reset();
476                self.radio.shorts.reset();
477                return Err(());
478            }
479        }
480    }

482    /// Sends the given `packet`
483    ///
484    /// This is utility method that *consecutively* calls the `try_send` method until it succeeds.
485    /// Note that this approach is *not* IEEE spec compliant -- there must be delay between failed
486    /// CCA attempts to be spec compliant
487    ///
488    /// NOTE this method will *not* modify the `packet` argument. The mutable reference is used to
489    /// ensure the `packet` buffer is allocated in RAM, which is required by the RADIO peripheral
490    // NOTE we do NOT check the address of `packet` because the mutable reference ensures it's
491    // allocated in RAM
492    pub fn send(&mut self, packet: &mut Packet) {
493        // enable radio to perform cca
494        self.put_in_rx_mode();
495
496        // clear related events
497        self.radio.events_phyend.reset();
498        self.radio.events_end.reset();
499
500        // immediately start transmission if the channel is idle
501        self.radio.shorts.modify(|_, w| {
502            w.ccaidle_txen()
503                .set_bit()
504                .txready_start()
505                .set_bit()
506                .end_disable()
507                .set_bit()
508        });
509
510        // the DMA transfer will start at some point after the following write operation so
511        // we place the compiler fence here
512        dma_start_fence();
513        // NOTE(unsafe) DMA transfer has not yet started
514        unsafe {
515            self.radio
516                .packetptr
517                .write(|w| w.packetptr().bits(packet.buffer.as_ptr() as u32));
518        }
519
520        'cca: loop {
521            // start CCA (+ sending if channel is clear)
522            self.radio
523                .tasks_ccastart
524                .write(|w| w.tasks_ccastart().set_bit());
525
526            loop {
527                if self.radio.events_phyend.read().events_phyend().bit_is_set() {
528                    dma_end_fence();
529                    // transmission is complete
530                    self.radio.events_phyend.reset();
531                    break 'cca;
532                }
533
534                if self
535                    .radio
536                    .events_ccabusy
537                    .read()
538                    .events_ccabusy()
539                    .bit_is_set()
540                {
541                    // channel is busy; try another CCA
542                    self.radio.events_ccabusy.reset();
543                    continue 'cca;
544                }
545            }
546        }
547
548        self.radio.shorts.reset();
549    }

551    /// Sends the specified `packet` without first performing CCA
552    ///
553    /// Acknowledgment packets must be sent using this method
554    ///
555    /// NOTE this method will *not* modify the `packet` argument. The mutable reference is used to
556    /// ensure the `packet` buffer is allocated in RAM, which is required by the RADIO peripheral
557    // NOTE we do NOT check the address of `packet` because the mutable reference ensures it's
558    // allocated in RAM
559    pub fn send_no_cca(&mut self, packet: &mut Packet) {
560        self.put_in_tx_mode();
561
562        // clear related events
563        self.radio.events_phyend.reset();
564        self.radio.events_end.reset();
565
566        // NOTE(unsafe) DMA transfer has not yet started
567        unsafe {
568            self.radio
569                .packetptr
570                .write(|w| w.packetptr().bits(packet.buffer.as_ptr() as u32));
571        }
572
573        // configure radio to disable transmitter once packet is sent
574        self.radio.shorts.modify(|_, w| w.end_disable().set_bit());
575
576        // start DMA transfer
577        dma_start_fence();
578        self.radio.tasks_start.write(|w| w.tasks_start().set_bit());
579
580        self.wait_for_event(Event::PhyEnd);
581        self.radio.shorts.reset();
582    }

584    /// Moves the radio from any state to the DISABLED state
585    fn disable(&mut self) {
586        // See figure 110 in nRF52840-PS
587        loop {
588            match self.radio.state.read().state().variant().unwrap() {
589                STATE_A::DISABLED => return,
590
591                STATE_A::RX_RU | STATE_A::RX_IDLE | STATE_A::TX_RU | STATE_A::TX_IDLE => {
592                    self.radio
593                        .tasks_disable
594                        .write(|w| w.tasks_disable().set_bit());
595
596                    self.wait_for_state_a(STATE_A::DISABLED);
597                    return;
598                }
599
600                // ramping down
601                STATE_A::RX_DISABLE | STATE_A::TX_DISABLE => {
602                    self.wait_for_state_a(STATE_A::DISABLED);
603                    return;
604                }
605
606                // cancel ongoing transfer or ongoing CCA
607                STATE_A::RX => {
608                    self.radio
609                        .tasks_ccastop
610                        .write(|w| w.tasks_ccastop().set_bit());
611                    self.radio.tasks_stop.write(|w| w.tasks_stop().set_bit());
612                    self.wait_for_state_a(STATE_A::RX_IDLE);
613                }
614                STATE_A::TX => {
615                    self.radio.tasks_stop.write(|w| w.tasks_stop().set_bit());
616                    self.wait_for_state_a(STATE_A::TX_IDLE);
617                }
618            }
619        }
620    }

622    /// Moves the radio to the RXIDLE state
623    fn put_in_rx_mode(&mut self) {
624        let state = self.state();
625
626        let (disable, enable) = match state {
627            State::Disabled => (false, true),
628            State::RxIdle => (false, self.needs_enable),
629            // NOTE to avoid errata 204 (see rev1 v1.4) we do TXIDLE -> DISABLED -> RXIDLE
630            State::TxIdle => (true, true),
631        };
632
633        if disable {
634            self.radio
635                .tasks_disable
636                .write(|w| w.tasks_disable().set_bit());
637            self.wait_for_state_a(STATE_A::DISABLED);
638        }
639
640        if enable {
641            self.needs_enable = false;
642            self.radio.tasks_rxen.write(|w| w.tasks_rxen().set_bit());
643            self.wait_for_state_a(STATE_A::RX_IDLE);
644        }
645    }
646
647    /// Moves the radio to the TXIDLE state
648    fn put_in_tx_mode(&mut self) {
649        let state = self.state();
650
651        if state != State::TxIdle || self.needs_enable {
652            self.needs_enable = false;
653            self.radio.tasks_txen.write(|w| w.tasks_txen().set_bit());
654            self.wait_for_state_a(STATE_A::TX_IDLE);
655        }
656    }
657
658    fn state(&self) -> State {
659        match self.radio.state.read().state().variant().unwrap() {
660            // final states
661            STATE_A::DISABLED => State::Disabled,
662            STATE_A::TX_IDLE => State::TxIdle,
663            STATE_A::RX_IDLE => State::RxIdle,
664
665            // transitory states
666            STATE_A::TX_DISABLE => {
667                self.wait_for_state_a(STATE_A::DISABLED);
668                State::Disabled
669            }
670
671            _ => unreachable!(),
672        }
673    }
674
675    fn wait_for_event(&self, event: Event) {
676        match event {
677            Event::End => {
678                while self.radio.events_end.read().events_end().bit_is_clear() {}
679                self.radio.events_end.reset();
680            }
681            Event::PhyEnd => {
682                while self
683                    .radio
684                    .events_phyend
685                    .read()
686                    .events_phyend()
687                    .bit_is_clear()
688                {}
689                self.radio.events_phyend.reset();
690            }
691        }
692    }
693
694    /// Waits until the radio state matches the given `state`
695    fn wait_for_state_a(&self, state: STATE_A) {
696        while self.radio.state.read().state().variant().unwrap() != state {}
697    }

700/// Error
701#[derive(Copy, Clone, Debug, PartialEq)]
702pub enum Error {
703    /// Incorrect CRC

735/// An IEEE 802.15.4 packet
736///
737/// This `Packet` is a PHY layer packet. It's made up of the physical header (PHR) and the PSDU
738/// (PHY service data unit). The PSDU of this `Packet` will always include the MAC level CRC, AKA
739/// the FCS (Frame Control Sequence) -- the CRC is fully computed in hardware and automatically
740/// appended on transmission and verified on reception.
741///
742/// The API lets users modify the usable part (not the CRC) of the PSDU via the `deref` and
743/// `copy_from_slice` methods. These methods will automatically update the PHR.
744///
745/// See figure 119 in the Product Specification of the nRF52840 for more details
746pub struct Packet {
747    buffer: [u8; Self::SIZE],
748}
749
750// See figure 124 in nRF52840-PS
751impl Packet {
752    // for indexing purposes
753    const PHY_HDR: usize = 0;
754    const DATA: RangeFrom<usize> = 1..;
755
756    /// Maximum amount of usable payload (CRC excluded) a single packet can contain, in bytes
757    pub const CAPACITY: u8 = 125;
758    const CRC: u8 = 2; // size of the CRC, which is *never* copied to / from RAM
759    const MAX_PSDU_LEN: u8 = Self::CAPACITY + Self::CRC;
760    const SIZE: usize = 1 /* PHR */ + Self::MAX_PSDU_LEN as usize;
761
762    /// Returns an empty packet (length = 0)
763    pub fn new() -> Self {
764        let mut packet = Self {
765            buffer: [0; Self::SIZE],
766        };
767        packet.set_len(0);
768        packet
769    }
770
771    /// Fills the packet payload with given `src` data
772    ///
773    /// # Panics
774    ///
775    /// This function panics if `src` is larger than `Self::CAPACITY`
776    pub fn copy_from_slice(&mut self, src: &[u8]) {
777        assert!(src.len() <= Self::CAPACITY as usize);
778        let len = src.len() as u8;
779        self.buffer[Self::DATA][..len as usize].copy_from_slice(&src[..len.into()]);
780        self.set_len(len);
781    }
782
783    /// Returns the size of this packet's payload
784    pub fn len(&self) -> u8 {
785        self.buffer[Self::PHY_HDR] - Self::CRC
786    }
787
788    /// Changes the size of the packet's payload
789    ///
790    /// # Panics
791    ///
792    /// This function panics if `len` is larger than `Self::CAPACITY`
793    pub fn set_len(&mut self, len: u8) {
794        assert!(len <= Self::CAPACITY);
795        self.buffer[Self::PHY_HDR] = len + Self::CRC;
796    }
797
798    /// Returns the LQI (Link Quality Indicator) of the received packet
799    ///
800    /// Note that the LQI is stored in the `Packet`'s internal buffer by the hardware so the value
801    /// returned by this method is only valid after a `Radio.recv` operation. Operations that
802    /// modify the `Packet`, like `copy_from_slice` or `set_len`+`deref_mut`, will overwrite the
803    /// stored LQI value.
804    ///
805    /// Also note that the hardware will *not* compute a LQI for packets smaller than 3 bytes so
806    /// this method will return an invalid value for those packets.
807    pub fn lqi(&self) -> u8 {
808        self.buffer[1 /* PHY_HDR */ + self.len() as usize /* data */]
809    }
810}
811
812impl ops::Deref for Packet {
813    type Target = [u8];
814
815    fn deref(&self) -> &[u8] {
816        &self.buffer[Self::DATA][..self.len() as usize]
817    }
818}
819
820impl ops::DerefMut for Packet {
821    fn deref_mut(&mut self) -> &mut [u8] {
822        let len = self.len();
823        &mut self.buffer[Self::DATA][..len as usize]
824    }
