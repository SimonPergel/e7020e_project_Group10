1//! A `usb-device` implementation using the USBD peripheral.
2//!
3//! Difficulties:
4//! * Control EP 0 is special:
5//!   * Setup stage is put in registers, not RAM.
6//!   * Different events are used to initiate transfers.
7//!   * No notification when the status stage is ACK'd.
8

45#[derive(Copy, Clone)]
46enum TransferState {
47    NoTransfer,

51#[derive(Copy, Clone)]
52struct EP0State {

54    remaining_size: u16,

59/// USB device implementation.
60///
61/// This type implements the [`UsbBus`] trait and can be passed to a [`UsbBusAllocator`] to
62/// configure and use the USB device.
63///
64/// [`UsbBusAllocator`]: usb_device::bus::UsbBusAllocator
65pub struct Usbd<T: UsbPeripheral> {
66    _periph: Mutex<T>,
67    // argument passed to `UsbDeviceBuilder.max_packet_size_0`
68    max_packet_size_0: u16,
69    bufs: Buffers,
70    used_in: u8,
71    used_out: u8,
72    iso_in_used: bool,
73    iso_out_used: bool,
74    ep0_state: Mutex<Cell<EP0State>>,
75    busy_in_endpoints: Mutex<Cell<u16>>,
76}

78impl<T: UsbPeripheral> Usbd<T> {
79    /// Creates a new USB device wrapper, taking ownership of the raw peripheral.
80    ///
81    /// # Parameters
82    ///
83    /// * `periph`: The raw USBD peripheral.
84    #[inline]
85    pub fn new(periph: T) -> Self {
86        Self {
87            _periph: Mutex::new(periph),
88            max_packet_size_0: 0,
89            bufs: Buffers::new(),
90            used_in: 0,
91            used_out: 0,
92            iso_in_used: false,
93            iso_out_used: false,
94            ep0_state: Mutex::new(Cell::new(EP0State {
95                direction: UsbDirection::Out,
96                remaining_size: 0,
97                in_transfer_state: TransferState::NoTransfer,
98                is_set_address: false,
99            })),
100            busy_in_endpoints: Mutex::new(Cell::new(0)),
101        }
102    }
103
104    fn regs<'a>(&self, _cs: &'a CriticalSection) -> &'a RegisterBlock {
105        unsafe { &*(T::REGISTERS as *const RegisterBlock) }
106    }
107
108    /// Fetches the address assigned to the device (only valid when device is configured).
109    pub fn device_address(&self) -> u8 {
110        unsafe { &*(T::REGISTERS as *const RegisterBlock) }
111            .usbaddr
112            .read()
113            .addr()
114            .bits()
115    }
116
117    fn is_used(&self, ep: EndpointAddress) -> bool {
118        if ep.index() == 8 {
119            // ISO
120            let flag = if ep.is_in() {
121                self.iso_in_used
122            } else {
123                self.iso_out_used
124            };
125            return flag;
126        }
127
128        if ep.is_in() {
129            self.used_in & (1 << ep.index()) != 0
130        } else {
131            self.used_out & (1 << ep.index()) != 0
132        }
133    }
134
135    fn read_control_setup(
136        &self,
137        regs: &RegisterBlock,
138        buf: &mut [u8],
139        ep0_state: &mut EP0State,
140    ) -> usb_device::Result<usize> {
141        const SETUP_LEN: usize = 8;
142
143        if buf.len() < SETUP_LEN {
144            return Err(UsbError::BufferOverflow);
145        }
146
147        // This is unfortunate: Reassemble the nicely split-up setup packet back into bytes, only
148        // for the usb-device code to copy the bytes back out into structured data.
149        // The bytes are split across registers, leaving a 3-Byte gap between them, so we couldn't
150        // even memcpy all of them at once. Weird peripheral.
151        buf[0] = regs.bmrequesttype.read().bits() as u8;
152        buf[1] = regs.brequest.read().brequest().bits();
153        buf[2] = regs.wvaluel.read().wvaluel().bits();
154        buf[3] = regs.wvalueh.read().wvalueh().bits();
155        buf[4] = regs.windexl.read().windexl().bits();
156        buf[5] = regs.windexh.read().windexh().bits();
157        buf[6] = regs.wlengthl.read().wlengthl().bits();
158        buf[7] = regs.wlengthh.read().wlengthh().bits();
159
160        ep0_state.direction = match regs.bmrequesttype.read().direction().is_host_to_device() {
161            false => UsbDirection::In,
162            true => UsbDirection::Out,
163        };
164        ep0_state.remaining_size = (buf[6] as u16) | ((buf[7] as u16) << 8);
165        ep0_state.is_set_address = (buf[0] == 0x00) && (buf[1] == 0x05);
166
167        if ep0_state.direction == UsbDirection::Out {
168            regs.tasks_ep0rcvout
169                .write(|w| w.tasks_ep0rcvout().set_bit());
170        }
171
172        Ok(SETUP_LEN)
173    }
174}

176impl<T: UsbPeripheral> UsbBus for Usbd<T> {
177    fn alloc_ep(
178        &mut self,
179        ep_dir: UsbDirection,
180        ep_addr: Option<EndpointAddress>,
181        ep_type: EndpointType,
182        max_packet_size: u16,
183        _interval: u8,
184    ) -> usb_device::Result<EndpointAddress> {
185        // Endpoint addresses are fixed in hardware:
186        // - 0x80 / 0x00 - Control        EP0
187        // - 0x81 / 0x01 - Bulk/Interrupt EP1
188        // - 0x82 / 0x02 - Bulk/Interrupt EP2
189        // - 0x83 / 0x03 - Bulk/Interrupt EP3
190        // - 0x84 / 0x04 - Bulk/Interrupt EP4
191        // - 0x85 / 0x05 - Bulk/Interrupt EP5
192        // - 0x86 / 0x06 - Bulk/Interrupt EP6
193        // - 0x87 / 0x07 - Bulk/Interrupt EP7
194        // - 0x88 / 0x08 - Isochronous
195
196        // Endpoint directions are allocated individually.
197
198        // store user-supplied value
199        if ep_addr.map(|addr| addr.index()) == Some(0) {
200            self.max_packet_size_0 = max_packet_size;
201        }
202
203        let (used, lens) = match ep_dir {
204            UsbDirection::In => (&mut self.used_in, &mut self.bufs.in_lens),
205            UsbDirection::Out => (&mut self.used_out, &mut self.bufs.out_lens),
206        };
207
208        let alloc_index = match ep_type {
209            EndpointType::Isochronous { .. } => {
210                let flag = match ep_dir {
211                    UsbDirection::In => &mut self.iso_in_used,
212                    UsbDirection::Out => &mut self.iso_out_used,
213                };
214
215                if *flag {
216                    return Err(UsbError::EndpointOverflow);
217                } else {
218                    *flag = true;
219                    lens[8] = max_packet_size as u8;
220                    return Ok(EndpointAddress::from_parts(0x08, ep_dir));
221                }
222            }
223            EndpointType::Control => 0,
224            EndpointType::Interrupt | EndpointType::Bulk => {
225                let leading = used.leading_zeros();
226                if leading == 0 {
227                    return Err(UsbError::EndpointOverflow);
228                }
229
230                if leading == 8 {
231                    // Even CONTROL is free, don't allocate that
232                    1
233                } else {
234                    8 - leading
235                }
236            }
237        };
238
239        if *used & (1 << alloc_index) != 0 {
240            return Err(UsbError::EndpointOverflow);
241        }
242
243        *used |= 1 << alloc_index;
244        lens[alloc_index as usize] = max_packet_size as u8;
245
246        let addr = EndpointAddress::from_parts(alloc_index as usize, ep_dir);
247        Ok(addr)
248    }

250    #[inline]
251    fn enable(&mut self) {
252        critical_section::with(move |cs| {
253            let regs = self.regs(&cs);
254
255            errata::pre_enable();
256
257            regs.enable.write(|w| w.enable().enabled());
258
259            // Wait until the peripheral is ready.
260            while !regs.eventcause.read().ready().is_ready() {}
261            regs.eventcause.write(|w| w.ready().set_bit()); // Write 1 to clear.
262
263            errata::post_enable();
264
265            // Enable the USB pullup, allowing enumeration.
266            regs.usbpullup.write(|w| w.connect().enabled());
267        });
268    }

270    #[inline]
271    fn reset(&self) {
272        critical_section::with(move |cs| {
273            let regs = self.regs(&cs);
274
275            // TODO: Initialize ISO buffers
276
277            // XXX this is not spec compliant; the endpoints should only be enabled after the device
278            // has been put in the Configured state. However, usb-device provides no hook to do that
279            // TODO: Merge `used_{in,out}` with `iso_{in,out}_used` so ISO is enabled here as well.
280            // Make the enabled endpoints respond to traffic.
281            unsafe {
282                regs.epinen.write(|w| w.bits(self.used_in.into()));
283                regs.epouten.write(|w| w.bits(self.used_out.into()));
284            }
285
286            for i in 1..8 {
287                let out_enabled = self.used_out & (1 << i) != 0;
288
289                // when first enabled, bulk/interrupt OUT endpoints will *not* receive data (the
290                // peripheral will NAK all incoming packets) until we write a zero to the SIZE
291                // register (see figure 203 of the 52840 manual). To avoid that we write a 0 to the
292                // SIZE register
293                if out_enabled {
294                    regs.size.epout[i].reset();
295                }
296            }
297
298            self.busy_in_endpoints.borrow(cs).set(0);
299        });
300    }

307    fn write(&self, ep_addr: EndpointAddress, buf: &[u8]) -> usb_device::Result<usize> {
308        if !self.is_used(ep_addr) {
309            return Err(UsbError::InvalidEndpoint);
310        }
311
312        if ep_addr.is_out() {
313            return Err(UsbError::InvalidEndpoint);
314        }
315
316        // A 0-length write to Control EP 0 is a status stage acknowledging a control write xfer
317        if ep_addr.index() == 0 && buf.is_empty() {
318            let exit = critical_section::with(move |cs| {
319                let regs = self.regs(&cs);
320
321                let ep0_state = self.ep0_state.borrow(cs).get();
322
323                if ep0_state.is_set_address {
324                    // Inhibit
325                    return true;
326                }
327
328                if ep0_state.direction == UsbDirection::Out {
329                    regs.tasks_ep0status
330                        .write(|w| w.tasks_ep0status().set_bit());
331                    return true;
332                }
333
334                if ep0_state.direction == UsbDirection::In && ep0_state.remaining_size == 0 {
335                    // Device sent all the requested data, no need to send ZLP.
336                    // Host will issue an OUT transfer in this case, device should
337                    // respond with a status stage.
338                    regs.tasks_ep0status
339                        .write(|w| w.tasks_ep0status().set_bit());
340                    return true;
341                }
342
343                false
344            });
345
346            if exit {
347                return Ok(0);
348            }
349        }
350
351        let i = ep_addr.index();
352
353        if usize::from(self.bufs.in_lens[i]) < buf.len() {
354            return Err(UsbError::BufferOverflow);
355        }
356
357        critical_section::with(move |cs| {
358            let regs = self.regs(&cs);
359            let busy_in_endpoints = self.busy_in_endpoints.borrow(cs);
360
361            if busy_in_endpoints.get() & (1 << i) != 0 {
362                // Maybe this endpoint is not busy?
363                let epdatastatus = regs.epdatastatus.read().bits();
364                if epdatastatus & (1 << i) != 0 {
365                    // Clear the event flag
366                    regs.epdatastatus.write(|w| unsafe { w.bits(1 << i) });
367
368                    // Clear the busy status and continue
369                    busy_in_endpoints.set(busy_in_endpoints.get() & !(1 << i));
370                } else {
371                    return Err(UsbError::WouldBlock);
372                }
373            }
374            if regs.epstatus.read().bits() & (1 << i) != 0 {
375                return Err(UsbError::WouldBlock);
376            }
377
378            let mut ram_buf: MaybeUninit<[u8; 64]> = MaybeUninit::uninit();
379            unsafe {
380                let slice = &mut *ram_buf.as_mut_ptr();
381                slice[..buf.len()].copy_from_slice(buf);
382            }
383            let ram_buf = unsafe { ram_buf.assume_init() };
384
385            let epin = [
386                &regs.epin0,
387                &regs.epin1,
388                &regs.epin2,
389                &regs.epin3,
390                &regs.epin4,
391                &regs.epin5,
392                &regs.epin6,
393                &regs.epin7,
394            ];
395
396            // Set the buffer length so the right number of bytes are transmitted.
397            // Safety: `buf.len()` has been checked to be <= the max buffer length.
398            unsafe {
399                if buf.is_empty() {
400                    epin[i].ptr.write(|w| w.bits(0));
401                } else {
402                    epin[i].ptr.write(|w| w.bits(ram_buf.as_ptr() as u32));
403                }
404                epin[i].maxcnt.write(|w| w.maxcnt().bits(buf.len() as u8));
405            }
406
407            if i == 0 {
408                // EPIN0: a short packet (len < max_packet_size0) indicates the end of the data
409                // stage and must be followed by us responding with an ACK token to an OUT token
410                // sent from the host (AKA the status stage) -- `usb-device` provides no call back
411                // for that so we'll trigger the status stage using a shortcut
412                let is_short_packet = buf.len() < self.max_packet_size_0 as usize;
413                regs.shorts.modify(|_, w| {
414                    if is_short_packet {
415                        w.ep0datadone_ep0status().set_bit()
416                    } else {
417                        w.ep0datadone_ep0status().clear_bit()
418                    }
419                });
420
421                let mut ep0_state = self.ep0_state.borrow(cs).get();
422                ep0_state.remaining_size =
423                    ep0_state.remaining_size.saturating_sub(buf.len() as u16);
424                self.ep0_state.borrow(cs).set(ep0_state);
425
426                // Hack: trigger status stage if the IN transfer is not acknowledged after a few frames,
427                // so record the current frame here; the actual test and status stage activation happens
428                // in the poll method.
429                let frame_counter = regs.framecntr.read().framecntr().bits();
430                let ep0_state = self.ep0_state.borrow(cs);
431                let mut state = ep0_state.get();
432                state.in_transfer_state = TransferState::Started(frame_counter);
433                ep0_state.set(state);
434            }
435
436            // Clear ENDEPIN[i] flag
437            regs.events_endepin[i].reset();
438
439            // Kick off device -> host transmission. This starts DMA, so a compiler fence is needed.
440            dma_start();
441            regs.tasks_startepin[i].write(|w| w.tasks_startepin().set_bit());
442            while regs.events_endepin[i]
443                .read()
444                .events_endepin()
445                .bit_is_clear()
446            {}
447            regs.events_endepin[i].reset();
448            dma_end();
449
450            // Clear EPSTATUS.EPIN[i] flag
451            regs.epstatus.write(|w| unsafe { w.bits(1 << i) });
452
453            // Mark the endpoint as busy
454            busy_in_endpoints.set(busy_in_endpoints.get() | (1 << i));
455
456            Ok(buf.len())
457        })
458    }

460    fn read(&self, ep_addr: EndpointAddress, buf: &mut [u8]) -> usb_device::Result<usize> {
461        if !self.is_used(ep_addr) {
462            return Err(UsbError::InvalidEndpoint);
463        }
464
465        if ep_addr.is_in() {
466            return Err(UsbError::InvalidEndpoint);
467        }
468
469        let i = ep_addr.index();
470        critical_section::with(move |cs| {
471            let regs = self.regs(&cs);
472
473            // Control EP 0 is special
474            if i == 0 {
475                // Control setup packet is special, since it is put in registers, not a buffer.
476                if regs.events_ep0setup.read().events_ep0setup().bit_is_set() {
477                    regs.events_ep0setup.reset();
478
479                    let ep0_state = self.ep0_state.borrow(cs);
480                    let mut state = ep0_state.get();
481                    let n = self.read_control_setup(regs, buf, &mut state)?;
482                    ep0_state.set(state);
483
484                    return Ok(n);
485                } else {
486                    // Is the endpoint ready?
487                    if regs
488                        .events_ep0datadone
489                        .read()
490                        .events_ep0datadone()
491                        .bit_is_clear()
492                    {
493                        // Not yet ready.
494                        return Err(UsbError::WouldBlock);
495                    }
496                }
497            } else {
498                // Is the endpoint ready?
499                let epdatastatus = regs.epdatastatus.read().bits();
500                if epdatastatus & (1 << (i + 16)) == 0 {
501                    // Not yet ready.
502                    return Err(UsbError::WouldBlock);
503                }
504            }
505
506            // Check that the packet fits into the buffer
507            let size = regs.size.epout[i].read().bits();
508            if size as usize > buf.len() {
509                return Err(UsbError::BufferOverflow);
510            }
511
512            // Clear status
513            if i == 0 {
514                regs.events_ep0datadone.reset();
515            } else {
516                regs.epdatastatus
517                    .write(|w| unsafe { w.bits(1 << (i + 16)) });
518            }
519
520            // We checked that the endpoint has data, time to read it
521
522            let epout = [
523                &regs.epout0,
524                &regs.epout1,
525                &regs.epout2,
526                &regs.epout3,
527                &regs.epout4,
528                &regs.epout5,
529                &regs.epout6,
530                &regs.epout7,
531            ];
532            epout[i]
533                .ptr
534                .write(|w| unsafe { w.bits(buf.as_ptr() as u32) });
535            // MAXCNT must match SIZE
536            epout[i].maxcnt.write(|w| unsafe { w.bits(size) });
537
538            dma_start();
539            regs.events_endepout[i].reset();
540            regs.tasks_startepout[i].write(|w| w.tasks_startepout().set_bit());
541            while regs.events_endepout[i]
542                .read()
543                .events_endepout()
544                .bit_is_clear()
545            {}
546            regs.events_endepout[i].reset();
547            dma_end();
548
549            // TODO: ISO
550
551            // Enable the endpoint
552            regs.size.epout[i].reset();
553
554            Ok(size as usize)
555        })
556    }

558    fn set_stalled(&self, ep_addr: EndpointAddress, stalled: bool) {
559        critical_section::with(move |cs| {
560            let regs = self.regs(&cs);
561
562            unsafe {
563                if ep_addr.index() == 0 {
564                    regs.tasks_ep0stall
565                        .write(|w| w.tasks_ep0stall().bit(stalled));
566                } else {
567                    regs.epstall.write(|w| {
568                        w.ep()
569                            .bits(ep_addr.index() as u8 & 0b111)
570                            .io()
571                            .bit(ep_addr.is_in())
572                            .stall()
573                            .bit(stalled)
574                    });
575                }
576            }
577
578            if stalled {
579                let busy_in_endpoints = self.busy_in_endpoints.borrow(cs);
580                busy_in_endpoints.set(busy_in_endpoints.get() & !(1 << ep_addr.index()));
581            }
582        });
583    }

585    fn is_stalled(&self, ep_addr: EndpointAddress) -> bool {
586        critical_section::with(move |cs| {
587            let regs = self.regs(&cs);
588
589            let i = ep_addr.index();
590            match ep_addr.direction() {
591                UsbDirection::Out => regs.halted.epout[i].read().getstatus().is_halted(),
592                UsbDirection::In => regs.halted.epin[i].read().getstatus().is_halted(),
593            }
594        })
595    }

616    fn poll(&self) -> PollResult {
617        critical_section::with(move |cs| {
618            let regs = self.regs(&cs);
619            let busy_in_endpoints = self.busy_in_endpoints.borrow(cs);
620
621            if regs.events_usbreset.read().events_usbreset().bit_is_set() {
622                regs.events_usbreset.reset();
623                return PollResult::Reset;
624            } else if regs.events_usbevent.read().events_usbevent().bit_is_set() {
625                // "Write 1 to clear"
626                if regs.eventcause.read().suspend().bit() {
627                    regs.eventcause.write(|w| w.suspend().bit(true));
628                    return PollResult::Suspend;
629                } else if regs.eventcause.read().resume().bit() {
630                    regs.eventcause.write(|w| w.resume().bit(true));
631                    return PollResult::Resume;
632                } else {
633                    regs.events_usbevent.reset();
634                }
635            }
636
637            if regs.events_sof.read().events_sof().bit_is_set() {
638                regs.events_sof.reset();
639
640                // Check if we have a timeout for EP0 IN transfer
641                let ep0_state = self.ep0_state.borrow(cs);
642                let mut state = ep0_state.get();
643                if let TransferState::Started(counter) = state.in_transfer_state {
644                    let frame_counter = regs.framecntr.read().framecntr().bits();
645                    if frame_counter.wrapping_sub(counter) >= 5 {
646                        // Send a status stage to ACK a pending OUT transfer
647                        regs.tasks_ep0status
648                            .write(|w| w.tasks_ep0status().set_bit());
649
650                        // reset the state
651                        state.in_transfer_state = TransferState::NoTransfer;
652                        ep0_state.set(state);
653                    }
654                }
655            }
656
657            // Check for any finished transmissions.
658            let mut in_complete = 0;
659            let mut out_complete = 0;
660            if regs
661                .events_ep0datadone
662                .read()
663                .events_ep0datadone()
664                .bit_is_set()
665            {
666                let ep0_state = self.ep0_state.borrow(cs).get();
667                if ep0_state.direction == UsbDirection::In {
668                    // Clear event, since we must only report this once.
669                    regs.events_ep0datadone.reset();
670
671                    in_complete |= 1;
672
673                    // Reset a timeout for the IN transfer
674                    let ep0_state = self.ep0_state.borrow(cs);
675                    let mut state = ep0_state.get();
676                    state.in_transfer_state = TransferState::NoTransfer;
677                    ep0_state.set(state);
678
679                    // Mark the endpoint as not busy
680                    busy_in_endpoints.set(busy_in_endpoints.get() & !1);
681                } else {
682                    // Do not clear OUT events, since we have to continue reporting them until the
683                    // buffer is read.
684
685                    out_complete |= 1;
686                }
687            }
688            let epdatastatus = regs.epdatastatus.read().bits();
689            for i in 1..=7 {
690                if epdatastatus & (1 << i) != 0 {
691                    // EPDATASTATUS.EPIN[i] is set
692
693                    // Clear event, since we must only report this once.
694                    regs.epdatastatus.write(|w| unsafe { w.bits(1 << i) });
695
696                    in_complete |= 1 << i;
697
698                    // Mark the endpoint as not busy
699                    busy_in_endpoints.set(busy_in_endpoints.get() & !(1 << i));
700                }
701                if epdatastatus & (1 << (i + 16)) != 0 {
702                    // EPDATASTATUS.EPOUT[i] is set
703                    // This flag will be cleared in `read()`
704
705                    out_complete |= 1 << i;
706                }
707            }
708
709            // Setup packets are only relevant on the control EP 0.
710            let mut ep_setup = 0;
711            if regs.events_ep0setup.read().events_ep0setup().bit_is_set() {
712                ep_setup = 1;
713
714                // Reset shorts
715                regs.shorts
716                    .modify(|_, w| w.ep0datadone_ep0status().clear_bit());
717            }
718
719            // TODO: Check ISO EP
720
721            if out_complete != 0 || in_complete != 0 || ep_setup != 0 {
722                PollResult::Data {
723                    ep_out: out_complete,
724                    ep_in_complete: in_complete,
725                    ep_setup,
726                }
727            } else {
728                PollResult::None
729            }
730        })
731    }
732
733    fn force_reset(&self) -> usb_device::Result<()> {
734        critical_section::with(move |cs| {
735            self.regs(&cs).usbpullup.write(|w| w.connect().disabled());
736        });
737
738        // Delay for 1ms, to give the host a chance to detect this.
739        // We run at 64 MHz, so 64k cycles are 1ms.
740        cortex_m::asm::delay(64_000);
741
742        critical_section::with(move |cs| {
743            self.regs(&cs).usbpullup.write(|w| w.connect().enabled());
744        });
745
746        Ok(())
747    }
