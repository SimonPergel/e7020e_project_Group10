69pub struct Visitor<T> {
70    pub inner: T,
71//    err: Option<syn::Error>,
72}
73
74impl<T> Visitor<T> {
75    pub fn new(inner: T) -> Self {
76        Self { inner }
77    }
78
79    pub fn process(&mut self, item: TokenStream2) -> syn::Result<TokenStream2>
80    where
81        Self: VisitMutExt,
82    {
83        let mut syntax_tree: syn::File = syn::parse(item.into())?;
84        self.visit_file_mut(&mut syntax_tree);
85        let ts = quote!(#syntax_tree);
86
87        Ok(ts)
88    }
89}

95macro_rules! impl_fn {
96    (@expr $value:expr, value) => {
97        $value
98    };
99    (@expr $value:expr, (_, value)) => {
100        &mut $value.1
101    };
102    (@func $self:expr, $node:ident, $proc:ident(node $(.$path:ident)? as Some($expr:tt) $(, $mode:expr)?) ) => {
103        if let Some(value) = impl_fn!(@arg $node $(.$path)?) {
104            $self.$proc( impl_fn!(@expr value, $expr) $(, $mode)? ).unwrap();
105        };
106    };
107    (@func $self:expr, $node:ident, $proc:ident(node $(.$path:ident)? $(, $mode:expr)?) ) => {
108        $self.$proc( impl_fn!(@arg $node $(.$path)?) $(, $mode)? ).unwrap();
109    };
110    (@funcs $self:expr, $node:ident, { $($proc:ident $params:tt ;)+ } ) => {
111        $(
112            impl_fn!(@func $self, $node, $proc $params);
113        )+
114    };
115    (@arg $node:ident $(.$path:ident)+) => {
116        &mut $node $(.$path)+
117    };
118    (@arg $node:ident) => {
119        $node
120    };
121    ($name:ident, $ty:ty, ) => {
122        fn $name(&mut self, node: &mut $ty) {
123            visit_mut::$name(self, node);
124        }
125    };
126    ($name:ident, $ty:ty, $before:tt) => {
127        fn $name(&mut self, node: &mut $ty) {
128            impl_fn!(@funcs self, node, $before);
129
130            visit_mut::$name(self, node);
131        }
132    };
133    ($name:ident, $ty:ty, $before:tt, $after:tt) => {
134        fn $name(&mut self, node: &mut $ty) {
135            impl_fn!(@funcs self, node, $before);
136
137            visit_mut::$name(self, node);
138
139            impl_fn!(@funcs self, node, $after);
140        }
141    };
142}
